<CodePlayground
  language="javascript"
  title="Manejo de objetos JSON"
  initialCode={`// --- Trabajando con JSON ---

// Un objeto JavaScript (como los que maneja el backend)
const tarea = {
  id: 1,
  titulo: "Estudiar APIs REST",
  completada: false,
  tags: ["backend", "express"]
};

// Acceder a propiedades
console.log("Titulo:", tarea.titulo);
console.log("Primer tag:", tarea.tags[0]);

// Convertir objeto a JSON string (lo que envia el servidor)
const jsonString = JSON.stringify(tarea, null, 2);
console.log("\\nJSON stringificado:");
console.log(jsonString);

// Convertir JSON string a objeto (lo que recibe el servidor)
const jsonRecibido = '{"titulo": "Leer apunte", "prioridad": "alta"}';
const objetoParseado = JSON.parse(jsonRecibido);
console.log("\\nObjeto parseado:", objetoParseado);
console.log("Prioridad:", objetoParseado.prioridad);

// Desestructuracion (como se usa en los controllers)
const { titulo, prioridad } = objetoParseado;
console.log("\\nDesestructurado - titulo:", titulo);
console.log("Desestructurado - prioridad:", prioridad);`}
/>

<CodePlayground
  language="javascript"
  title="Simulando Request y Response"
  initialCode={`// --- Simulacion de objetos req y res de Express ---

// Simulamos el objeto req (request) que llega al handler
const req = {
  method: "POST",
  url: "/api/tareas",
  params: { id: "5" },
  query: { orden: "fecha", completada: "true" },
  body: { titulo: "Estudiar Node.js", descripcion: "Leer el apunte completo" }
};

console.log("=== Peticion recibida ===");
console.log("Metodo:", req.method, req.url);

// Acceder a parametros de ruta (req.params)
console.log("\\nParametro de ruta (id):", req.params.id);

// Acceder a query strings (req.query)
console.log("Query - orden:", req.query.orden);
console.log("Query - completada:", req.query.completada);

// Acceder al body (req.body) con desestructuracion
const { titulo, descripcion } = req.body;
console.log("\\nBody - titulo:", titulo);
console.log("Body - descripcion:", descripcion);

// Simulamos la logica de un controller
console.log("\\n=== Respuesta del servidor ===");

// Validacion (como en un controller real)
if (!titulo || titulo.trim() === "") {
  console.log("Status: 400");
  console.log(JSON.stringify({ error: "El titulo es obligatorio" }));
} else {
  // Simulamos crear el recurso
  const nuevaTarea = {
    id: 3,
    titulo: titulo.trim(),
    descripcion: descripcion || null,
    completada: false
  };
  console.log("Status: 201 Created");
  console.log(JSON.stringify(nuevaTarea, null, 2));
}`}
/>
