// "Base de datos" en memoria
const usuarios = [];

// === Funciones CRUD ===

function crearUsuario(body) {
  if (!body.nombre || !body.email) {
    return { status: 400, body: { error: "nombre y email son obligatorios" } };
  }
  const nuevo = { id: nextId++, nombre: body.nombre, email: body.email };

  return { status: 201, body: nuevo };
}

function obtenerTodos() {
  return { status: 200, body: [...usuarios] };
}

function obtenerPorId(id) {
  const u = usuarios.find((u) => u.id === id);
  if (!u) return { status: 404, body: { error: "No encontrado" } };
  return { status: 200, body: u };
}

function actualizar(id, body) {
  const idx = usuarios.findIndex((u) => u.id === id);
  if (idx === -1) return { status: 404, body: { error: "No encontrado" } };
  if (body.nombre) usuarios[idx].nombre = body.nombre;

// Funcion que valida un objeto usuario y retorna un array de errores

function validarUsuario(usuario) {
  const errores = [];

  // Validar nombre: debe ser un string no vacio
  if (!usuario.nombre || typeof usuario.nombre !== "string") {
    errores.push("'nombre' es obligatorio y debe ser un string");
  } else if (usuario.nombre.trim() === "") {
    errores.push("'nombre' no puede estar vacio");
  }

  // Validar email: debe ser string y contener @
  if (!usuario.email || typeof usuario.email !== "string") {
    errores.push("'email' es obligatorio y debe ser un string");
  } else if (!usuario.email.includes("@")) {
    errores.push("'email' debe contener @");
  }

  // Validar edad: debe ser un numero positivo
  if (usuario.edad === undefined || usuario.edad === null) {
    errores.push("'edad' es obligatorio");
  } else if (typeof usuario.edad !== "number") {
    errores.push("'edad' debe ser un numero");
  } else if (usuario.edad <= 0) {
    errores.push("'edad' debe ser un numero positivo");

}

function loggerMiddleware(req, res, next) {
  const timestamp = new Date().toISOString();
  console.log(\`[\${timestamp}] \${req.method} \${req.url}\`);
  console.log("  -> Paso por middleware: Logger");

// Verifica que el request tenga un token de autorizacion
function authMiddleware(req, res, next) {
  console.log("  -> Paso por middleware: Auth Check");

  const authHeader = req.headers["Authorization"];
  if (!authHeader || !authHeader.startsWith("Bearer ")) {
    console.log("  !! Auth FALLIDA: sin token");
    res.status(401).json({ error: "No autorizado" });
    return; // No llama a next(), corta la cadena
  }

  console.log(\`  -> Token encontrado: \${token.substring(0, 10)}...\`);
  // En la realidad aca se verificaria el token (JWT, etc.)
  req.usuario = { id: 1, nombre: "Ana", rol: "admin" }; // Datos del token
  next(); // Token valido, seguir
